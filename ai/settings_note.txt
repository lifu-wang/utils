lifu@PCHOME:~$ sudo install -D -o root -g root -m 644 packages.microsoft.gpg /usr/share/keyrings/packages.microsoft.gpg
lifu@PCHOME:~$ echo "deb [arch=amd64 signed-by=/usr/share/keyrings/packages.microsoft.gpg] https://packages.microsoft.com/repos/code stable main" | sudo tee /etc/apt/sources.list.d/vscode.list
lifu@PCHOME:~$ sudo apt update
lifu@PCHOME:~$ sudo apt install code -y
